57  	 */
58  	private void initializeGame() {
59  		setVictory(false);
60  
61  		try {
62  			model.generateMap(LevelSelector.getLevel());
63  		} catch (SQLException e) {
64  			e.printStackTrace();
65  		}
66  		gameController = new GameController(model.getLorann(), this, instantiateDemonMover());
67  		game = new Thread(gameController);
68  		gameFrame.setVisible(true);
69  	}

71  	/**
72  	 * Instantiate the movement of the demon
73  	 * 
74  	 * @return movers The list of all the movers
75  	 */
76  	private CopyOnWriteArrayList<DemonMover> instantiateDemonMover() {
77  		CopyOnWriteArrayList<Demon> demons = ComponentFactory.getDemons();
78  		CopyOnWriteArrayList<DemonMover> movers = new CopyOnWriteArrayList<DemonMover>();
79  
80  		// Loop which add a new DemonMover for each demon into the ArrayList
81  		for (Demon demon : demons) {
82  			movers.add(new DemonMover(demon, this));
83  		}
84  
85  		return movers;
86  	}

88  	/**
89  	 * Method used to move a component
90  	 * 
91  	 * @param component
92  	 *            The component to move
93  	 * @param direction
94  	 *            The direction that the component will take
95  	 */
96  	@Override
97  	public synchronized void moveComponent(IComponent component, Direction direction) {
98  		ICoordinate currentCoordinates = component.getCoordinate();
99  		ICoordinate newCoordinates;
100 
101 		switch (direction) {
102 		case DOWN:
103 			newCoordinates = new Coordinate(currentCoordinates.getX() + 1, currentCoordinates.getY());
104 			break;
105 
106 		case DOWNLEFT:
107 			newCoordinates = new Coordinate(currentCoordinates.getX() + 1, currentCoordinates.getY() - 1);
108 			break;
109 
110 		case DOWNRIGHT:
111 			newCoordinates = new Coordinate(currentCoordinates.getX() + 1, currentCoordinates.getY() + 1);
112 			break;
113 
114 		case LEFT:
115 			newCoordinates = new Coordinate(currentCoordinates.getX(), currentCoordinates.getY() - 1);
116 			break;
117 
118 		case RIGHT:
119 			newCoordinates = new Coordinate(currentCoordinates.getX(), currentCoordinates.getY() + 1);
120 			break;
121 
122 		case UP:
123 			newCoordinates = new Coordinate(currentCoordinates.getX() - 1, currentCoordinates.getY());
124 			break;
125 
126 		case UPLEFT:
127 			newCoordinates = new Coordinate(currentCoordinates.getX() - 1, currentCoordinates.getY());
128 			break;
129 
130 		case UPRIGHT:
131 			newCoordinates = new Coordinate(currentCoordinates.getX() - 1, currentCoordinates.getY() + 1);
132 			break;
133 		default:
134 			newCoordinates = new Coordinate(currentCoordinates.getX(), currentCoordinates.getY());
135 			break;
136 		}
137 
138 		// Condition: if the component is Lorann, check if Lorann can move
139 		if (component instanceof Lorann) {
140 			component.setDirection(direction);
141 			if (directionisAvailableForLorann(newCoordinates)) {
142 				checkTargetLocation(component, model.getMap()[newCoordinates.getX()][newCoordinates.getY()]);
143 
144 				component.setCoordinate(newCoordinates);
145 				model.getMap()[currentCoordinates.getX()][currentCoordinates.getY()] = new Empty(currentCoordinates);
146 				model.getMap()[newCoordinates.getX()][newCoordinates.getY()] = component;
147 			}
148 
149 			// Condition: if the component is a demon, check if it can move
150 		} else if (component instanceof Demon) {
151 			if (directionIsAvailableForDemons(newCoordinates)) {
152 				checkTargetLocation(component, model.getMap()[newCoordinates.getX()][newCoordinates.getY()]);
153 
154 				component.setCoordinate(newCoordinates);
155 				component.setDirection(direction);
156 				model.getMap()[currentCoordinates.getX()][currentCoordinates.getY()] = new Empty(currentCoordinates);
157 				model.getMap()[newCoordinates.getX()][newCoordinates.getY()] = component;
158 			}
159 
160 			// Condition: if the component is the spell, move it
161 		} else if (component instanceof Spell) {
162 			moveSpell((Spell) component, newCoordinates, direction);
163 		}
164 	}

166 	/**
167 	 * This method is used to move the <b>spell</b>. Because the spell move in a
168 	 * very different way than others components, we use a different method than
169 	 * <b>moveComponent()</b>.
170 	 * 
171 	 * @param spell
172 	 *            the spell to move
173 	 * @param newCoordinatesForTheSpell
174 	 *            the future coordinates of the spell
175 	 * @param the
176 	 *            current direction of the spell
177 	 */
178 	private synchronized void moveSpell(Spell spell, ICoordinate newCoordinatesForTheSpell,
179 			Direction currentDirection) {
180 
181 		// If the future position of the spell contain Lorann
182 		if (model.getMap()[newCoordinatesForTheSpell.getX()][newCoordinatesForTheSpell.getY()] instanceof Lorann) {
183 			spell.setTarget(model.getMap()[newCoordinatesForTheSpell.getX()][newCoordinatesForTheSpell.getY()]);
184 			spell.actionWhenContactHappend();
185 			model.getMap()[spell.getCoordinate().getX()][spell.getCoordinate().getY()] = new Empty(
186 					spell.getCoordinate());
187 			model.setSpellToNull();
188 
189 			// If the future position of the spell contain a demon
190 		} else if (model.getMap()[newCoordinatesForTheSpell.getX()][newCoordinatesForTheSpell
191 				.getY()] instanceof Demon) {
192 			spell.setTarget(model.getMap()[newCoordinatesForTheSpell.getX()][newCoordinatesForTheSpell.getY()]);
193 			spell.actionWhenContactHappend();
194 
195 			// If the future position of the spell contain a border
196 		} else if (model.getMap()[newCoordinatesForTheSpell.getX()][newCoordinatesForTheSpell
197 				.getY()] instanceof Setting) {
198 			spell.setDirection(Direction.getOppositeDirection(currentDirection));
199 			moveComponent(spell, spell.getDirection());
200 
201 			// If the future position of the spell contain an energy sphere
202 		} else if ((model.getMap()[newCoordinatesForTheSpell.getX()][newCoordinatesForTheSpell
203 				.getY()] instanceof EnergySphere)
204 				|| (model.getMap()[newCoordinatesForTheSpell.getX()][newCoordinatesForTheSpell
205 						.getY()] instanceof Gate)) {
206 			spell.setDirection(Direction.getOppositeDirection(currentDirection));
207 			moveComponent(spell, spell.getDirection());
208 
209 		} else {
210 			if (!(model.getMap()[spell.getCoordinate().getX()][spell.getCoordinate().getY()] instanceof Lorann)) {
211 				model.getMap()[spell.getCoordinate().getX()][spell.getCoordinate().getY()] = new Empty(
212 						spell.getCoordinate());
213 			}
214 			spell.setCoordinate(newCoordinatesForTheSpell);
215 			model.getMap()[newCoordinatesForTheSpell.getX()][newCoordinatesForTheSpell.getY()] = spell;
216 
217 		}
218 	}

221 	 * Check if Lorann can move here.
222 	 * 
223 	 * @param coordinateToCheck
224 	 * @return a boolean to allow or not the move
225 	 */
226 	private synchronized boolean directionisAvailableForLorann(ICoordinate coordinateToCheck) {
227 		IComponent destination = model.getMap()[coordinateToCheck.getX()][coordinateToCheck.getY()];
228 
229 		if (destination.isLORANN_PERMEABLE()) {
230 			return true;
231 		} else {
232 			return false;
233 		}

253 	/**
254 	 * Realize action relative to the target component
255 	 * 
256 	 * @param componentToMove
257 	 * @param componentInPosition
258 	 */
259 	private synchronized void checkTargetLocation(IComponent componentToMove, IComponent componentInPosition) {
260 		if (componentToMove instanceof Lorann) {
261 			Lorann lorann = (Lorann) componentToMove;
262 			// If Lorann catch a treasure, give 10 points to the player.
263 			if (componentInPosition instanceof Treasure) {
264 				Treasure treasure = (Treasure) componentInPosition;
265 
266 				this.score = score + treasure.getValue();
267 
268 				// If Lorann meet a demon, kill him
269 			} else if (componentInPosition instanceof Demon) {
270 				this.model.getLorann().kill();
271 
272 			} else if (componentInPosition instanceof Gate) {
273 				Gate gate = (Gate) componentInPosition;
274 				// If gate is available, we win the game
275 				if (gate.isAvailable()) {
276 					setVictory(true);
277 				} else {
278 					// If the gate is not available, kill Lorann
279 					lorann.kill();
280 				}
281 
282 			} else if (componentInPosition instanceof EnergySphere) {
283 				EnergySphere sphere = (EnergySphere) componentInPosition;
284 				// When we take the EnergySphere, open the gate
285 				if (sphere.isAvailable()) {
286 					model.getGate().setAvailable(true);
287 				}
288 				sphere.setAvailable(false);
289 
290 				// If the position of Lorann will be the spell, get it
291 			} else if (componentInPosition instanceof Spell) {
292 				lorann.setSpellLaunched(false);
293 				model.getMap()[componentInPosition.getCoordinate().getX()][componentInPosition.getCoordinate()
294 						.getY()] = new Empty(componentInPosition.getCoordinate());
295 				model.setSpellToNull();
296 
297 			}
298 
299 			// If Lorann has the same position than a demon, Lorann die.
300 		} else if (componentToMove instanceof Demon) {
301 			if (componentInPosition instanceof Lorann) {
302 				Lorann lorann = (Lorann) componentInPosition;
303 				lorann.kill();
304 
305 			} else if (componentInPosition instanceof Spell) {
306 				Demon demon = (Demon) componentToMove;
307 				demon.kill();
308 			}
309 		}
310 	}

317 	/**
318 	 * Set an action for Lorann
319 	 * 
320 	 * @param order
321 	 *            The order Lorann has to execute
322 	 */
323 	public void setAction(Order order) {
324 		if (order.equals(Order.FIRE)) {
325 			// Launch the spell if the order is to fire
326 			if (model.getLorann().launchSpell()) {
327 				moveComponent(
328 						(IComponent) model.getSpell(new Coordinate(model.getLorann().getCoordinate()),
329 								Direction.getOppositeDirection(model.getLorann().getDirection())),
330 						Direction.getOppositeDirection(model.getLorann().getDirection()));
331 			}
332 		} else {
333 			// Move the component
334 			moveComponent(model.getLorann(), Direction.getDirectionFromOrder(order));
335 		}
336 
337 	}
