16  @SuppressWarnings("serial")
17  public class GamePanel extends JPanel implements IGamePanel, KeyListener {
18  	private IComponent[][] components;
19  	private Observer observer;
20  
21  	public GamePanel(Model model, Observer observer) {
22  		this.addKeyListener(this);
23  		this.observer = observer;
24  		this.setBackground(Color.BLACK);
25  	}
26  
27  	public void updateMap(IComponent[][] components) {
28  		/*We update the map and we repaint*/
29  		this.components = components;
30  		repaint();
31  	}
32  
33  	@Override
34  	protected void paintComponent(Graphics g) {
35  		super.paintComponent(g);
36  			/*We enlarge the images to put the game in full screen*/
37  		for (int x = 0; x < components.length; x++) {
38  			for (int y = 0; y < components[x].length; y++) {
39  				g.drawImage(components[x][y].getImage(Direction.LEFT), y * 91, x * 91, 80, 80, null);
40  			}
41  		}
42  
43  	}
44  
45  	@Override
46  	public void keyPressed(KeyEvent e) {
47  		int keyCode = e.getKeyCode();
48  		/*we inform our program of the action of certain keys on the keyboard */
49  		switch (keyCode) {
50  		case KeyEvent.VK_UP:
51  			observer.update(null, Direction.UP);
52  			break;
53  
54  		case KeyEvent.VK_DOWN:
55  			observer.update(null, Direction.DOWN);
56  			break;
57  
58  		case KeyEvent.VK_LEFT:
59  			observer.update(null, Direction.LEFT);
60  			break;
61  
62  		case KeyEvent.VK_RIGHT:
63  			observer.update(null, Direction.RIGHT);
64  			break;
65  
66  		case KeyEvent.VK_ESCAPE:
67  			System.exit(0);
68  			break;
69  
70  		case KeyEvent.VK_SPACE:
71  			// observer.update();
72  		}
73  
74  	}
