27  	@Override
28  	public void kill(Component target) {
29  		if (!isAvailable()) {
30  			if (target instanceof Lorann) {
31  				Lorann character = (Lorann) target;
32  				character.kill();
33  			}
34  		}
35  	}
36  /**
37   * If direction equal up, the sprite of the door is closed else, it's
38   * open
39   */
40  	@Override
41  	public Image getImage(Direction direction) { 
42  													 
43  		if (isAvailable()) {
44  			return getSprites().get(1);
45  		} else {
46  			return getSprites().get(0);
47  		}
48  	}
